interface FrozenPolicySnapshot {
  policyVersionId: string;
  createdAt: string;

  basePredictionRule: string;
  confidenceRule: string;
  escalationRule: string;
  uncertaintyRule: string;

  calibrationWindow: number;

  boundedAdjustmentState: {
    confidenceBias: number;
    escalationBias: number;
    uncertaintyBias: number;
  };

  calibrationState: {
    totalPredictions: number;
    resolvedPredictions: number;
    exactMatchRate: number | null;
    directionMatchRate: number | null;
    tooAggressiveRate: number | null;
    tooWeakRate: number | null;
  };

  explanationLines: string[];
}

interface FrozenPredictionRecord {
  sourceVersionId: string;
  predictionTimestamp: string;

  predictedVariable: string | null;
  confidence: "low" | "moderate" | "high";
  riskDirection: "decreasing" | "stable" | "rising";

  frozenPolicyVersionId: string;
  frozenPolicySnapshot: FrozenPolicySnapshot;
}

Why this matters

Without this, future NOMOS can say:
	•	“I predicted X”

But not:
	•	“I predicted X under policy regime Y”

That means calibration history becomes blurry.

With policy freeze, you can later compare:
	•	predictions made under policy v1
	•	predictions made under policy v2
	•	whether policy changes improved calibration
	•	whether bounded adjustment helped or hurt

That is a much more serious system.

What this unlocks next

Once policy freeze exists, you can do:
	•	calibration by policy version
	•	accuracy comparison across policy regimes
	•	rollback if a new policy performs worse
	•	exact replay of old predictions

That is a huge win.

What comes after this

After policy freeze + versioning, the strongest next step is:

policy regime comparison

So NOMOS can say:
	•	policy v2 reduced over-aggressive predictions
	•	policy v3 increased uncertainty appropriately under shallow history
	•	policy v1 had better exact-match rate for nutrition audits

That would make the system not just reproducible, but evaluable across its own evolution.
