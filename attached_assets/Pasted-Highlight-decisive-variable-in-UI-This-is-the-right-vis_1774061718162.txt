🧩 2. Update StatusCard.tsx

📁 StatusCard.tsx

Replace basic line rendering with:

function renderLine(line: string, decisive?: string) {
  if (!decisive) return line;

  const lower = line.toLowerCase();
  const key = decisive.toLowerCase();

  if (!lower.includes(key)) return line;

  const parts = line.split(new RegExp(`(${key})`, "i"));

  return parts.map((part, i) => {
    if (part.toLowerCase() === key) {
      return (
        <span key={i} className="nm-decisive">
          {part}
        </span>
      );
    }
    return <span key={i}>{part}</span>;
  });
}


⸻

Use it in render:

<div className="nm-status-body">
  {message.lines.map((line, i) => (
    <div key={i} className="nm-line">
      {renderLine(line, message.decisiveVariable)}
    </div>
  ))}
</div>
