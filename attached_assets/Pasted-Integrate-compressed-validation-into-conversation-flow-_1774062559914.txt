🧠 2. Build unified system line

📁 conversation_formatter.ts

import { compressIssues } from "../validation/validation_format";

export function buildConversationLine(
  state: ConversationState
): string {
  const issues = state.issues ?? [];

  // If there are validation issues → lead with them
  if (issues.length > 0) {
    const compressed = compressIssues(issues);

    // pick the highest priority domain line
    const primary = compressed[0];

    const nextPrompt = getPrompt(state);

    return `${primary}\n${nextPrompt}`;
  }

  // No issues → just prompt
  return getPrompt(state);
}

🧠 3. Stage-aware prompt (tightened)

Update your prompt function to be sharper:

function getPrompt(state: ConversationState): string {
  switch (state.stage) {
    case "INTENT":
      return "State objective precisely.";

    case "CONSTRAINTS":
      return "Declare constraints.";

    case "MODEL_ASSUMPTIONS":
      return "Specify assumptions.";

    case "CONFIRMATION":
      return "Confirm submission or revise.";

    default:
      return "";
  }
}

🧠 4. Full conversation step

export function stepConversation(
  state: ConversationState,
  input: string
): ConversationOutput {
  // 1. update draft
  const updated = updateDraft(state, input);

  // 2. detect issues
  const issues = detectIssues(updated.draftQuery);

  // 3. compute completeness
  const completeness = computeCompleteness(
    updated.draftQuery,
    issues
  );

  // 4. determine next stage
  const next = nextStage({
    ...updated,
    issues,
  });

  const nextState: ConversationState = {
    ...updated,
    stage: next,
    issues,
    completeness,
  };

  // 5. build output line
  const line = buildConversationLine(nextState);

  return {
    line,
    draft: nextState.draftQuery,
    issues,
    completeness,
  };
}
