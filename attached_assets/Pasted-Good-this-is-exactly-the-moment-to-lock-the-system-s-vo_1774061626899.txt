🧠 2. Enforce Line Budget

Replace your body builder with:

function enforceLineBudget(
  status: "LAWFUL" | "DEGRADED" | "INVALID",
  lines: string[]
): string[] {
  const max = MAX_LINES[status];
  return lines.slice(0, max);
}


⸻

🧠 3. Enforce Character Width

function enforceLineWidth(lines: string[]): string[] {
  return lines.map((line) => {
    if (line.length <= MAX_CHARS_PER_LINE) return line;

    return line.slice(0, MAX_CHARS_PER_LINE - 3) + "...";
  });
}

🧠 4. Final Output Sanitizer (CRITICAL)

function finalizeOutput(
  status: "LAWFUL" | "DEGRADED" | "INVALID",
  rawLines: string[]
): string[] {
  let lines = rawLines
    .filter(Boolean)
    .map((l) => sanitizePhrase(l));

  lines = enforceLineBudget(status, lines);
  lines = enforceLineWidth(lines);

  return lines;
}
