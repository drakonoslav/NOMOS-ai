🧠 2. Guidance generator

import { ValidationIssue } from "./validation_engine";

export function generateCompletenessGuidance(
  draft: Partial<NomosQuery>,
  issues: ValidationIssue[],
  breakdown: CompletenessResult["breakdown"]
): string | undefined {
  // Priority 1: blocking errors (most specific first)
  const thresholdErr = issues.find((i) =>
    i.id.includes("threshold")
  );
  if (thresholdErr) {
    return "Completeness limited by missing constraint thresholds.";
  }

  const unitWarn = issues.find((i) =>
    i.id.includes("unit")
  );
  if (unitWarn) {
    return "Completeness limited by missing units.";
  }

  const ambiguousWarn = issues.find((i) =>
    i.id.includes("ambiguous")
  );
  if (ambiguousWarn) {
    return "Completeness limited by ambiguous constraints.";
  }

  // Priority 2: structural gaps
  if (!breakdown.intent) {
    return "Completeness limited by undefined intent.";
  }

  if (!breakdown.constraintsPresent) {
    return "Completeness limited by missing constraints.";
  }

  if (!breakdown.constraintsValid) {
    return "Completeness limited by invalid constraints.";
  }

  if (!breakdown.assumptions) {
    return "Completeness limited by missing assumptions.";
  }

  // Fully complete
  if (breakdown.noErrors) {
    return "Submission structurally complete.";
  }

  return;
}

🧠 3. Integrate into scoring

Update computeCompleteness

export function computeCompleteness(
  draft: Partial<NomosQuery>,
  issues: ValidationIssue[]
): CompletenessResult {
  // ...existing logic...

  const result: CompletenessResult = {
    score,
    breakdown,
  };

  result.guidance = generateCompletenessGuidance(
    draft,
    issues,
    breakdown
  );

  return result;
}

🧠 Why this is correct

You achieve:
	•	explanation without verbosity
	•	guidance without suggestion overload
	•	clarity without breaking tone

⸻

🧠 Design constraints (important)

Guidance must:
	•	be one line only
	•	never include “you”
	•	never suggest actions directly
	•	always describe what is missing, not what to do
