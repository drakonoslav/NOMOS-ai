/**
 * src/ui/runtime/evaluate_nomos.ts
 *
 * Full NOMOS pipeline adapter:
 * belief → observer → model → proposer → decision → verification → guard → audit
 *
 * Produces a DashboardScenarioState for UI consumption.
 */

import { DashboardScenarioState } from "../demo/scenario_builder";
import { NomosQuery } from "../query/types";

// Core kernel modules (your existing files)
import { BeliefState } from "../../core/belief_state";
import { Observer } from "../../core/observer";
import { ModelRegistry } from "../../core/model_registry";
import { LLMProposer } from "../../core/llm_proposer";
import { DecisionEngine } from "../../core/decision_engine";
import { VerificationKernel } from "../../core/verification_kernel";
import { ConstitutionGuard } from "../../core/constitution_guard";
import { AuditLog } from "../../core/audit_log";

/* =========================================================
   Public API
   ========================================================= */

export async function runNomosEvaluation(
  query: NomosQuery
): Promise<DashboardScenarioState> {
  const audit = new AuditLog();

  /* -----------------------------------------
     1. BELIEF
  ----------------------------------------- */
  const belief = new BeliefState();
  belief.ingestQuery(query);

  audit.record("belief_initialized", {
    state: belief.getStateSnapshot(),
  });

  /* -----------------------------------------
     2. OBSERVER
  ----------------------------------------- */
  const observer = new Observer();
  const observation = observer.observe(belief);

  audit.record("observation_complete", observation);

  /* -----------------------------------------
     3. MODEL SELECTION
  ----------------------------------------- */
  const modelRegistry = new ModelRegistry();
  const model = modelRegistry.selectModel(observation);

  audit.record("model_selected", {
    modelId: model.id,
    confidence: model.confidence,
  });

  /* -----------------------------------------
     4. PROPOSER (LLM)
  ----------------------------------------- */
  const proposer = new LLMProposer();

  const proposalBundle = await proposer.propose({
    belief,
    observation,
    model,
  });

  audit.record("proposals_generated", {
    count: proposalBundle.proposals.length,
  });

  /* -----------------------------------------
     5. DECISION
  ----------------------------------------- */
  const decisionEngine = new DecisionEngine();

  const decision = decisionEngine.select({
    proposals: proposalBundle.proposals,
    model,
    belief,
  });

  audit.record("decision_selected", {
    selected: decision.selected?.id,
    rejected: decision.rejected.map((p) => p.id),
  });

  /* -----------------------------------------
     6. VERIFICATION
  ----------------------------------------- */
  const verificationKernel = new VerificationKernel();

  const verification = verificationKernel.verify({
    belief,
    decision,
    model,
    observation,
  });

  audit.record("verification_complete", verification);

  /* -----------------------------------------
     7. CONSTITUTION GUARD
  ----------------------------------------- */
  const guard = new ConstitutionGuard();

  const guarded = guard.enforce({
    verification,
    decision,
  });

  audit.record("constitution_enforced", guarded);

  /* -----------------------------------------
     8. FINAL AUTHORITY + ACTION
  ----------------------------------------- */
  const authority = guarded.authority; // AUTHORIZED | CONSTRAINED | REFUSED

  let actionOutcome: DashboardScenarioState["metrics"]["actionOutcome"];

  if (authority === "AUTHORIZED") {
    actionOutcome = "APPLIED";
  } else if (authority === "CONSTRAINED") {
    actionOutcome = "DEGRADED_ACTION_APPLIED";
  } else {
    actionOutcome = "REFUSED";
  }

  /* -----------------------------------------
     9. MAP → UI STATE (CRITICAL)
  ----------------------------------------- */

  const verificationStatus = mapVerificationStatus(verification);

  const selectedIds = decision.selected
    ? [decision.selected.id]
    : [];

  const rejectedIds = decision.rejected.map((p) => p.id);

  const scenarioLabel = buildLiveLabel(verificationStatus);

  const dashboardState: DashboardScenarioState = {
    scenario: mapScenarioKey(verificationStatus),
    label: scenarioLabel,
    description: "Live NOMOS evaluation of submitted query.",

    toneInput: {
      verificationStatus,
      authority,

      epsilonX: belief.getUncertainty(), // or derived metric
      identifiability: mapIdentifiability(verification),
      modelConfidence: model.confidence,

      robustnessEpsilon: verification.robustness?.epsilon,
      robustnessEpsilonMin: verification.robustness?.min,

      feasibilityOk: verification.feasibility.ok,
      robustnessOk: verification.robustness?.ok,
      observabilityOk: verification.observability?.ok,
      identifiabilityOk: verification.identifiability?.ok,
      modelOk: verification.model?.ok,
      adaptationOk: verification.adaptation?.ok,

      selectedCandidateIds: selectedIds,
      rejectedCandidateIds: rejectedIds,

      activeConstraint: findActiveConstraint(verification),
      decisiveVariable: findDecisiveVariable(verification),

      reasons: verification.reasons ?? [],
      adjustments: guarded.adjustments ?? [],
    },

    metrics: {
      selectedPlanId: decision.selected?.id,
      proposalCount: proposalBundle.proposals.length,
      actionOutcome,
    },
  };

  audit.record("runtime_complete", {
    status: verificationStatus,
    authority,
  });

  return dashboardState;
}

🔧 Helper mappers (same file)

function mapVerificationStatus(
  verification: any
): "LAWFUL" | "DEGRADED" | "INVALID" {
  if (!verification.feasibility.ok) return "INVALID";
  if (verification.robustness?.ok === false) return "DEGRADED";
  if (verification.model?.ok === false) return "DEGRADED";
  return "LAWFUL";
}

function mapScenarioKey(
  status: "LAWFUL" | "DEGRADED" | "INVALID"
) {
  if (status === "LAWFUL") return "lawful_baseline";
  if (status === "DEGRADED") return "degraded_low_margin";
  return "refused_infeasible";
}

function buildLiveLabel(
  status: "LAWFUL" | "DEGRADED" | "INVALID"
) {
  return `Live Evaluation (${status})`;
}

function mapIdentifiability(v: any): "FULL" | "PARTIAL" | "NONE" {
  if (!v.identifiability) return "NONE";
  if (v.identifiability.ok) return "FULL";
  return "PARTIAL";
}

function findActiveConstraint(v: any): string {
  return v?.feasibility?.violatedConstraint ?? "none";
}

function findDecisiveVariable(v: any): string {
  if (!v) return "unknown";

  if (!v.feasibility.ok) return "feasibility constraint";
  if (v.robustness?.ok === false) return "robustness margin";
  if (v.model?.ok === false) return "model confidence";

  return "robustness margin";
}
