🧠 3. Suggestion generators (deterministic)

function suggestThreshold(text: string): string {
  const t = text.toLowerCase();

  if (t.includes("sleep")) return "sleep ≥ 7h";
  if (t.includes("hrv")) return "HRV ≥ baseline";
  if (t.includes("weight")) return "weight ≤ target";

  return "Specify threshold using ≥, ≤, or numeric bound.";
}

function suggestUnit(text: string): string {
  if (text.includes("sleep")) return "sleep ≥ 7h";
  if (text.includes("time")) return "time ≤ 60 min";

  return "Add unit (h, min, kg, %).";
}

function suggestClarification(text: string): string {
  return "Replace vague term with measurable variable.";
}

🧠 4. Convert validation → suggestions

📁 suggestion_engine.ts

Add:

export function validationToSuggestions(
  issues: ValidationIssue[]
): Suggestion[] {
  return issues
    .filter((i) => i.suggestion)
    .map((i) => ({
      id: `fix_${i.id}`,
      text: i.suggestion!,
      type: i.field,
      confidence: "high",
    }));
}

🧠 5. Merge with existing suggestion flow

export async function getSuggestions(
  draft: Partial<NomosQuery>,
  stage: string,
  input: string,
  issues: ValidationIssue[]
): Promise<Suggestion[]> {
  const rule = ruleBasedSuggestions(draft);
  const val = validationToSuggestions(issues);
  const llm = await generateRefinementSuggestions(stage, input);

  return dedupeSuggestions([...val, ...rule, ...llm]);
}

🧩 6. UI: pair issue + fix

📁 ValidationPanel.tsx (upgrade)

export function ValidationPanel({
  issues,
  onApply,
}: {
  issues: ValidationIssue[];
  onApply: (text: string) => void;
}) {
  if (!issues.length) return null;

  return (
    <div className="nm-validation">
      {issues.map((i) => (
        <div key={i.id} className="nm-validation-item">
          <div className={`nm-${i.severity}`}>
            {i.message}
          </div>

          {i.suggestion && (
            <div className="nm-fix">
              Suggested: {i.suggestion}
              <button onClick={() => onApply(i.suggestion!)}>
                Apply
              </button>
            </div>
          )}
        </div>
      ))}
    </div>
  );
}
