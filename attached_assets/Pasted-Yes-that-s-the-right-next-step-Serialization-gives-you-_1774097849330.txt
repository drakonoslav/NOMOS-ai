audit_store.ts

import { AuditRecord } from "./audit_types";

const STORAGE_KEY = "nomos_audit_history_v1";

export function listAuditRecords(): AuditRecord[] {
  const raw = localStorage.getItem(STORAGE_KEY);
  if (!raw) return [];

  try {
    const parsed = JSON.parse(raw) as AuditRecord[];
    return Array.isArray(parsed) ? parsed : [];
  } catch {
    return [];
  }
}

export function saveAuditRecord(record: AuditRecord): void {
  const records = listAuditRecords();
  const existingIndex = records.findIndex((r) => r.id === record.id);

  if (existingIndex >= 0) {
    records[existingIndex] = record;
  } else {
    records.unshift(record);
  }

  localStorage.setItem(STORAGE_KEY, JSON.stringify(records));
}

export function getAuditRecord(id: string): AuditRecord | null {
  const records = listAuditRecords();
  return records.find((r) => r.id === id) ?? null;
}

export function deleteAuditRecord(id: string): void {
  const records = listAuditRecords().filter((r) => r.id !== id);
  localStorage.setItem(STORAGE_KEY, JSON.stringify(records));
}

export function clearAuditRecords(): void {
  localStorage.removeItem(STORAGE_KEY);
}

Add helper:

function buildAuditRecord(params: {
  canonicalDeclaration: string;
  compileResult: any;
  patchedDraft: any;
  evaluationResult: any;
  isConfirmed: boolean;
}): AuditRecord | null {
  if (!draft) return null;

  return {
    id: activeAuditId ?? buildAuditId(),
    versionId: buildVersionId(),
    parentVersionId: activeVersionId ?? null,
    timestamp: new Date().toISOString(),
    intent: draft.intent,
    title: draft.title,
    isEvaluable: draft.isEvaluable,
    isConfirmed: params.isConfirmed,
    canonicalDeclaration: params.canonicalDeclaration,
    compileResult: compileResult ?? null,
    patchedDraft: params.patchedDraft ?? null,
    evaluationResult: params.evaluationResult ?? null,
  };
}

Update confirm:

function handleConfirm() {
  if (!draft?.isEvaluable) return;
  setIsConfirmed(true);

  const canonical = buildSerializedDraftRecord(draft);
  const record = buildAuditRecord({
    canonicalDeclaration: canonical.canonicalText,
    compileResult,
    patchedDraft: draft,
    evaluationResult: null,
    isConfirmed: true,
  });

  if (record) {
    saveAuditRecord(record);
    setActiveAuditId(record.id);
    setActiveVersionId(record.versionId);
  }
}

Update evaluate:

function handleEvaluate() {
  if (!draft || !draft.isEvaluable || !isConfirmed) return;

  const canonical = buildSerializedDraftRecord(draft);

  const evaluationResult = {
    status: "PENDING_RUNTIME",
    payload: null,
  };

  const record = buildAuditRecord({
    canonicalDeclaration: canonical.canonicalText,
    compileResult,
    patchedDraft: draft,
    evaluationResult,
    isConfirmed: true,
  });

  if (record) {
    saveAuditRecord(record);
    setActiveAuditId(record.id);
    setActiveVersionId(record.versionId);
  }

  console.log("Evaluating canonical draft:", canonical);
}

Optional AuditHistoryPanel.tsx

import React from "react";
import { AuditRecord } from "../../../audit/audit_types";

export function AuditHistoryPanel({
  records,
  onLoad,
}: {
  records: AuditRecord[];
  onLoad: (record: AuditRecord) => void;
}) {
  if (!records.length) return null;

  return (
    <div className="nm-audit-history">
      <div className="nm-audit-history__title">AUDIT HISTORY</div>

      <div className="nm-audit-history__list">
        {records.map((record) => (
          <button
            key={record.versionId}
            type="button"
            className="nm-audit-history__item"
            onClick={() => onLoad(record)}
          >
            <div className="nm-audit-history__item-title">{record.title}</div>
            <div className="nm-audit-history__item-meta">
              {record.intent} · {record.timestamp} · {record.versionId}
            </div>
          </button>
        ))}
      </div>
    </div>
  );
}
