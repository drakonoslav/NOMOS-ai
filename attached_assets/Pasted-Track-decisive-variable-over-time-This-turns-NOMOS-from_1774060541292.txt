🧩 2. Extract time series

📁 src/ui/audit/audit_timeseries.ts

import { AuditEntry } from "../../core/audit_log";

export interface RunSummary {
  timestamp: number;
  status: "LAWFUL" | "DEGRADED" | "INVALID";
  decisiveVariable: string;
  modelConfidence?: number;
  robustness?: number;
  feasibility?: boolean;
}

export function extractRunSummaries(
  entries: AuditEntry[]
): RunSummary[] {
  return entries
    .filter((e) => e.stage === "run_summary")
    .map((e) => ({
      timestamp: e.timestamp,
      status: e.payload.status,
      decisiveVariable: e.payload.decisiveVariable,
      modelConfidence: e.payload.modelConfidence,
      robustness: e.payload.robustness,
      feasibility: e.payload.feasibility,
    }))
    .sort((a, b) => a.timestamp - b.timestamp);
}

🧠 3. Detect drift vs stability

export function analyzeDrift(series: RunSummary[]) {
  if (series.length < 2) return { drift: false };

  const last = series[series.length - 1];
  const prev = series[series.length - 2];

  const variableChanged =
    last.decisiveVariable !== prev.decisiveVariable;

  const statusChanged = last.status !== prev.status;

  return {
    drift: variableChanged || statusChanged,
    variableChanged,
    statusChanged,
  };
}

📊 4. DecisiveTrendPanel.tsx

📁 src/ui/components/nomos/DecisiveTrendPanel.tsx

import React from "react";
import { useScenario } from "../../state/ScenarioContext";
import {
  extractRunSummaries,
  analyzeDrift,
} from "../../audit/audit_timeseries";

export function DecisiveTrendPanel() {
  const { auditEntries } = useScenario();

  const series = extractRunSummaries(auditEntries);
  const drift = analyzeDrift(series);

  return (
    <div className="panel trend-panel">
      <div className="panel-header">
        Decisive Variable Trend
      </div>

      {series.length === 0 && (
        <div className="trend-empty">
          No run history available.
        </div>
      )}

      {series.length > 0 && (
        <>
          <div className="trend-status">
            {drift.drift ? (
              <span className="trend-drift">
                Drift detected
              </span>
            ) : (
              <span className="trend-stable">
                Stable
              </span>
            )}
          </div>

          <div className="trend-list">
            {series.map((point, i) => (
              <div key={i} className="trend-item">
                <div className="trend-time">
                  {formatTime(point.timestamp)}
                </div>

                <div className="trend-variable">
                  {point.decisiveVariable}
                </div>

                <div
                  className={`trend-status-pill trend-status-${point.status.toLowerCase()}`}
                >
                  {point.status}
                </div>
              </div>
            ))}
          </div>
        </>
      )}
    </div>
  );
}
