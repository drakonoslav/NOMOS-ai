🧩 1. Conversation State Machine

📁 conversation_engine.ts

type Stage =
  | "INTENT"
  | "CONSTRAINTS"
  | "MODEL_ASSUMPTIONS"
  | "CONFIRMATION"
  | "EVALUATION";

interface ConversationState {
  stage: Stage;
  draftQuery: Partial<NomosQuery>;
}


⸻

🧠 2. Controlled prompts per stage

function getPrompt(state: ConversationState): string {
  switch (state.stage) {
    case "INTENT":
      return "State the objective precisely.";

    case "CONSTRAINTS":
      return "Declare constraints that must not be violated.";

    case "MODEL_ASSUMPTIONS":
      return "Specify assumptions or unknowns.";

    case "CONFIRMATION":
      return "Confirm submission or revise.";

    default:
      return "";
  }
}

🧠 3. User input → structured extraction

Use your existing LLM parser, but scoped:

function updateDraft(
  state: ConversationState,
  input: string
): ConversationState {
  // call parser with stage context
  const parsed = parseByStage(input, state.stage);

  return {
    ...state,
    draftQuery: {
      ...state.draftQuery,
      ...parsed,
    },
  };
}

🧠 4. Stage progression logic

function nextStage(state: ConversationState): Stage {
  if (!state.draftQuery.intent) return "INTENT";
  if (!state.draftQuery.constraints) return "CONSTRAINTS";
  if (!state.draftQuery.assumptions) return "MODEL_ASSUMPTIONS";
  return "CONFIRMATION";
}
