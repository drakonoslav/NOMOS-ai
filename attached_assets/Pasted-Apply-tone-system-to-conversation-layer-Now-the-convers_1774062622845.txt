Add conversation resolver

export function resolveConversationTone(input: {
  issues: ValidationIssue[];
  stage: Stage;
  decisiveVariable?: string;
}): ToneOutput {
  const { issues, stage } = input;

  // ---- STATUS ----
  const hasErrors = issues.some(i => i.severity === "error");

  const status = hasErrors ? "INVALID" : "INCOMPLETE";

  // ---- DIAGNOSTIC LINE ----
  const compressed = compressIssues(issues);
  const diagnostic = compressed[0]; // primary line

  // ---- PROMPT LINE ----
  const prompt = getPromptByStage(stage);

  // ---- EMPHASIS ----
  const emphasizedDiagnostic = emphasizeKeyTerms(diagnostic);

  return {
    status,
    lines: [emphasizedDiagnostic, prompt],
  };
}

🧠 2. Add emphasis engine

📁 tone_emphasis.ts

export function emphasizeKeyTerms(line: string): string {
  if (!line) return "";

  return line
    .replace("missing threshold", "**missing threshold**")
    .replace("ambiguous", "**ambiguous**")
    .replace("missing unit", "**missing unit**")
    .replace("undefined intent", "**undefined intent**");
}


⸻

🧠 3. Stage prompt mapping (tightened)

function getPromptByStage(stage: Stage): string {
  switch (stage) {
    case "INTENT":
      return "State objective precisely.";

    case "CONSTRAINTS":
      return "Declare constraints.";

    case "MODEL_ASSUMPTIONS":
      return "Specify assumptions.";

    case "CONFIRMATION":
      return "Confirm submission or revise.";

    default:
      return "";
  }
}

🧠 4. Replace conversation formatter

📁 conversation_formatter.ts

import { resolveConversationTone } from "../tone/tone_resolver";

export function buildConversationOutput(
  state: ConversationState
): ToneOutput {
  return resolveConversationTone({
    issues: state.issues ?? [],
    stage: state.stage,
  });
}

🧩 5. UI rendering (supports emphasis)

function renderLine(line: string) {
  // simple markdown bold parser
  return line.split("**").map((part, i) =>
    i % 2 === 1 ? (
      <strong key={i}>{part}</strong>
    ) : (
      part
    )
  );
}

<div className="nm-system-line">
  {output.lines.map((l, i) => (
    <div key={i}>{renderLine(l)}</div>
  ))}
