Absolutely. Here is a tiny scenario_builder.ts that emits three deterministic dashboard states:
	•	lawful_baseline
	•	degraded_low_margin
	•	refused_infeasible

These are designed so your StatusCard and VerificationSummary can render immediately, without waiting for the full runtime pipeline.

src/ui/demo/scenario_builder.ts

/**
 * src/ui/demo/scenario_builder.ts
 *
 * Tiny deterministic scenario builder for NOMOS dashboard rendering.
 *
 * Purpose:
 * - Provide immediate, stable demo states for UI pages
 * - Render LAWFUL / DEGRADED / INVALID without requiring live runtime execution
 * - Feed directly into tone_resolver-driven components
 */

import { ToneResolverInput } from "../tone/tone_resolver";

export type DemoScenario =
  | "lawful_baseline"
  | "degraded_low_margin"
  | "refused_infeasible";

export interface DashboardMetricState {
  selectedPlanId?: string;
  proposalCount: number;
  actionOutcome: "APPLIED" | "DEGRADED_ACTION_APPLIED" | "REFUSED";
}

export interface DashboardScenarioState {
  scenario: DemoScenario;
  label: string;
  description: string;
  toneInput: ToneResolverInput;
  metrics: DashboardMetricState;
}

export function buildScenarioState(
  scenario: DemoScenario
): DashboardScenarioState {
  switch (scenario) {
    case "lawful_baseline":
      return lawfulBaselineScenario();

    case "degraded_low_margin":
      return degradedLowMarginScenario();

    case "refused_infeasible":
      return refusedInfeasibleScenario();
  }
}

export function buildAllScenarioStates(): DashboardScenarioState[] {
  return [
    buildScenarioState("lawful_baseline"),
    buildScenarioState("degraded_low_margin"),
    buildScenarioState("refused_infeasible"),
  ];
}

Example usage

In a page or story:

import React, { useState } from "react";
import {
  buildScenarioState,
  DemoScenario,
} from "../demo/scenario_builder";
import { StatusCard } from "../components/nomos/StatusCard";
import { VerificationSummary } from "../components/nomos/VerificationSummary";

export function ScenarioPreviewPage() {
  const [scenario, setScenario] = useState<DemoScenario>("lawful_baseline");
  const state = buildScenarioState(scenario);

  return (
    <div style={{ display: "grid", gap: 16 }}>
      <div>
        <button onClick={() => setScenario("lawful_baseline")}>Lawful</button>
        <button onClick={() => setScenario("degraded_low_margin")}>Degraded</button>
        <button onClick={() => setScenario("refused_infeasible")}>Invalid</button>
      </div>

      <div>
        <h2>{state.label}</h2>
        <p>{state.description}</p>
      </div>

      <StatusCard input={state.toneInput} />
      <VerificationSummary input={state.toneInput} />
    </div>
  );
}


⸻

Why this is useful

This gives you immediate contrast:
	•	Lawful shows compressed, calm authorization
	•	Degraded shows constrained, reasoned limitation
	•	Invalid shows decisive refusal

So the UI can demonstrate NOMOS’s full behavioral range before the deeper runtime integration is finished.
