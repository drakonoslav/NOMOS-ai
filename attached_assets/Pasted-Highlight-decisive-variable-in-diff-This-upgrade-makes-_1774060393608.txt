🧠 2. Add decisive variable detection

export function diffRunsWithDecision(
  runA: AuditEntry[],
  runB: AuditEntry[]
): DiffResult {
  const stageDiffs = diffRuns(runA, runB);

  const beforeVerification = findStage(runA, "verification_complete");
  const afterVerification = findStage(runB, "verification_complete");

  const decisive = findDecisiveChange(
    beforeVerification,
    afterVerification
  );

  return {
    stageDiffs,
    decisive,
  };
}

🔍 3. Core logic: find decisive variable

function findDecisiveChange(a: any, b: any): DecisiveDiff | undefined {
  if (!a || !b) return;

  // 1. Feasibility (highest priority)
  if (a.feasibility?.ok !== b.feasibility?.ok) {
    return {
      variable: "feasibility",
      before: a.feasibility?.ok,
      after: b.feasibility?.ok,
      reason: "Feasibility transition determines legality boundary",
    };
  }

  // 2. Robustness
  if (a.robustness?.ok !== b.robustness?.ok) {
    return {
      variable: "robustness margin",
      before: a.robustness?.epsilon,
      after: b.robustness?.epsilon,
      reason: "Robustness crossing threshold reduces allowable operation",
    };
  }

  // 3. Model confidence
  if (a.model?.confidence !== b.model?.confidence) {
    const delta = Math.abs(
      (a.model?.confidence ?? 0) - (b.model?.confidence ?? 0)
    );

    if (delta > 0.1) {
      return {
        variable: "model confidence",
        before: a.model?.confidence,
        after: b.model?.confidence,
        reason: "Model reliability shift alters admissibility",
      };
    }
  }

  // 4. Identifiability
  if (a.identifiability?.ok !== b.identifiability?.ok) {
    return {
      variable: "identifiability",
      before: a.identifiability?.ok,
      after: b.identifiability?.ok,
      reason: "Loss of identifiability breaks system grounding",
    };
  }

  // 5. Observability
  if (a.observability?.ok !== b.observability?.ok) {
    return {
      variable: "observability",
      before: a.observability?.ok,
      after: b.observability?.ok,
      reason: "Insufficient observability limits control validity",
    };
  }

  // 6. Adaptation
  if (a.adaptation?.ok !== b.adaptation?.ok) {
    return {
      variable: "adaptation integrity",
      before: a.adaptation?.ok,
      after: b.adaptation?.ok,
      reason: "Adaptation breakdown prevents recovery",
    };
  }

  return;
}

function findStage(entries: AuditEntry[], stage: string) {
  return entries.find((e) => e.stage === stage)?.payload;
}
