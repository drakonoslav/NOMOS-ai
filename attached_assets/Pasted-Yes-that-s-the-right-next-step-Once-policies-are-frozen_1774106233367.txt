interface PolicyRegimeMetrics {
  policyVersionId: string;

  totalPredictions: number;
  resolvedPredictions: number;

  exactMatchRate: number | null;
  directionMatchRate: number | null;

  tooWeakRate: number | null;

  averageEscalationBias: number | null;
  averageUncertaintyBias: number | null;

  trainingPredictionCount: number;
  schedulePredictionCount: number;

interface PolicyRegimeComparison {
  beforePolicyVersionId: string;
  afterPolicyVersionId: string;

  exactMatchDelta: number | null;
  directionMatchDelta: number | null;
  tooAggressiveDelta: number | null;
  tooWeakDelta: number | null;

  summaryLines: string[];
  changed: boolean;
}

interface PolicyRegimeComparisonReport {
  regimes: PolicyRegimeMetrics[];
  pairwiseComparisons: PolicyRegimeComparison[];
  bestByExactMatch: string | null;
  bestByDirectionMatch: string | null;
  lowestAggressiveRate: string | null;
  summaryLines: string[];
}

What this should let NOMOS say

Examples:
	•	“Policy v2 reduced over-aggressive predictions by 14% relative to v1.”
	•	“Policy v3 improved direction-match accuracy but slightly reduced exact-match accuracy.”
	•	“Policy v1 remains strongest for nutrition exact-match performance.”
	•	“Policy v3 appears more conservative under shallow history, with higher uncertainty and fewer false escalations.”

That is the right level of self-evaluation.

Why this matters

Without regime comparison, frozen policies are just archived snapshots.

With regime comparison, they become:
	•	measurable
	•	comparable
	•	historically meaningful

That means NOMOS can evaluate not just:
	•	predictions

but:
	•	the policy that generated those predictions

That is a much more serious architecture.

The key discipline

Still do not let NOMOS auto-select the best regime yet.

First let it:
	•	measure
	•	compare
	•	explain

Only after that should you think about policy selection or promotion.

What comes after this

After policy regime comparison, the strongest next step is:

manual policy promotion + rollback

So you can explicitly choose:
	•	promote v3 to active policy
	•	rollback to v2 if v3 performs worse
	•	keep domain-specific policy preferences

That would make NOMOS not just historically evaluable, but operationally governable.
