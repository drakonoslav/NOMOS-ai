🧠 2. Core compression engine

function semanticCompress(
  lines: string[],
  status: "LAWFUL" | "DEGRADED" | "INVALID"
): string[] {
  if (lines.length <= 1) return lines;

  let result: string[] = [];

  for (let i = 0; i < lines.length; i++) {
    const current = lines[i];
    const next = lines[i + 1];

    if (next && canMerge(current, next)) {
      result.push(mergeLines(current, next));
      i++; // skip next
    } else {
      result.push(current);
    }
  }

  return result;
}

🧠 3. Merge detection

function canMerge(a: string, b: string): boolean {
  const A = normalize(a);
  const B = normalize(b);

  // Same subject domain
  if (shareKeyword(A, B, "robustness")) return true;
  if (shareKeyword(A, B, "feasibility")) return true;
  if (shareKeyword(A, B, "model")) return true;

  // Decisive + condition pairing
  if (A.includes("feasibility holds") && B.includes("decisive")) return true;

  // Prediction pairing
  if (A.includes("decisive") && B.includes("drift")) return true;

  return false;
}

🧠 4. Merge strategy (STRICT)

function mergeLines(a: string, b: string): string {
  const A = stripPeriod(a);
  const B = stripPeriod(b);

  // Special: feasibility + decisive
  if (A.includes("Feasibility holds") && B.includes("decisive")) {
    return `${A}; ${B.toLowerCase()}.`;
  }

  // Special: decisive + prediction
  if (A.includes("decisive") && B.includes("Given current drift")) {
    return `${A}; ${B.toLowerCase()}`;
  }

  // Default safe merge
  return `${A}; ${lowerFirst(B)}.`;
}

🧠 5. Helpers

function normalize(s: string): string {
  return s.toLowerCase();
}

function shareKeyword(a: string, b: string, k: string): boolean {
  return a.includes(k) && b.includes(k);
}

function stripPeriod(s: string): string {
  return s.replace(/\.$/, "");
}

function lowerFirst(s: string): string {
  return s.charAt(0).toLowerCase() + s.slice(1);
}
