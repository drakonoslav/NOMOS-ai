🧱 1. First — extend AuditLog to support persistence

📁 src/core/audit_log.ts (patch)

export interface AuditEntry {
  id: string;
  timestamp: number;
  stage: string;
  payload: any;
}

export class AuditLog {
  private entries: AuditEntry[] = [];

  record(stage: string, payload: any) {
    const entry: AuditEntry = {
      id: crypto.randomUUID(),
      timestamp: Date.now(),
      stage,
      payload,
    };

    this.entries.push(entry);
    this.persist();
  }

  getEntries(): AuditEntry[] {
    return this.entries;
  }

  clear() {
    this.entries = [];
    localStorage.removeItem("nomos_audit_log");
  }

  /* -----------------------------
     Persistence
  ----------------------------- */

  private persist() {
    try {
      localStorage.setItem(
        "nomos_audit_log",
        JSON.stringify(this.entries)
      );
    } catch {}
  }

  load() {
    try {
      const raw = localStorage.getItem("nomos_audit_log");
      if (!raw) return;

      this.entries = JSON.parse(raw);
    } catch {}
  }
}

📁 3. AuditTimelinePanel.tsx

import React, { useState } from "react";
import { useScenario } from "../../state/ScenarioContext";

export function AuditTimelinePanel() {
  const { auditEntries, clearAudit } = useScenario();
  const [expandedId, setExpandedId] = useState<string | null>(null);

  return (
    <div className="panel audit-panel">
      <div className="panel-header">
        Audit Timeline
        <button onClick={clearAudit} className="audit-clear">
          Clear
        </button>
      </div>

      <div className="audit-list">
        {auditEntries.length === 0 && (
          <div className="audit-empty">
            No audit entries yet.
          </div>
        )}

        {auditEntries.map((entry) => {
          const expanded = expandedId === entry.id;

          return (
            <div
              key={entry.id}
              className={`audit-item audit-item--${entry.stage}`}
            >
              <div
                className="audit-item__header"
                onClick={() =>
                  setExpandedId(expanded ? null : entry.id)
                }
              >
                <span className="audit-stage">
                  {formatStage(entry.stage)}
                </span>

                <span className="audit-time">
                  {formatTime(entry.timestamp)}
                </span>
              </div>

              {expanded && (
                <pre className="audit-payload">
                  {safeStringify(entry.payload)}
                </pre>
              )}
            </div>
          );
        })}
      </div>
    </div>
  );
}
