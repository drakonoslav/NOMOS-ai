ConstraintClass

Purpose

Defines the kind of admissibility boundary being enforced.

export type ConstraintClass =
  | "PROHIBITION"            // forbidden event/state must not occur
  | "MIN_THRESHOLD"          // value must be >= threshold
  | "MAX_THRESHOLD"          // value must be <= threshold
  | "RANGE"                  // value must stay within [min, max]
  | "CONTINUITY"             // max interruption/gap bound
  | "DEADLINE"               // event must occur before/after time boundary
  | "SEQUENCING"             // event ordering must hold
  | "RESOURCE_PRESERVATION"  // stock/integrity must remain preserved/non-negative
  | "COMPATIBILITY"          // coupled constraints must jointly hold
  | "UNCERTAINTY_SENSITIVE"; // admissibility depends on epistemic quality


⸻

EvaluationOperator

Purpose

Defines the formal comparison or logical test applied to a derived variable.

export type EvaluationOperator =
  | "FORBIDS"         // forbidden event present?
  | "GTE"             // >=
  | "LTE"             // <=
  | "BETWEEN"         // lower <= x <= upper
  | "MAX_GAP_LTE"     // largest gap <= threshold
  | "MIN_GAP_GTE"     // smallest gap >= threshold
  | "BEFORE"          // event A before boundary/event B
  | "AFTER"           // event A after boundary/event B
  | "WITHIN_WINDOW"   // occurs inside window
  | "ORDERED_BEFORE"  // event A must precede event B
  | "NON_NEGATIVE"    // resource/state >= 0
  | "PRESERVED"       // integrity/structure preserved
  | "JOINTLY_SATISFIED" // multiple coupled predicates satisfied
  | "CONFIDENCE_GTE"; // model/epistemic confidence >= threshold

DerivedVariable

Purpose

Represents a value NOMOS must compute from a candidate before evaluation.

export interface DerivedVariable<TValue = number | boolean | string> {
  /**
   * Stable identifier used in evaluation and UI.
   * Example: total_sleep_minutes, longest_wake_gap, drop_risk, soil_disturbance_score
   */
  key: string;

  /**
   * Human-readable label.
   */
  label: string;

  /**
   * Value derived from candidate state/action description.
   */
  value: TValue;

  /**
   * Units if numeric or temporal.
   * Examples: "minutes", "hours", "kg", "%", "score"
   */
  units?: string;

  /**
   * Where this value came from.
   * Example: ["sleep_interval_1", "wake_gap_1"]
   */
  provenance?: string[];

  /**
   * Whether this variable was computed deterministically or estimated.
   */
  derivationMode: "DETERMINISTIC" | "ESTIMATED" | "HYBRID";

  /**
   * Optional confidence if not purely deterministic.
   */
  confidence?: number;

  /**
   * Optional explanation of derivation.
   */
  notes?: string[];
}

AtomicConstraint

Purpose

Represents one fully decomposed evaluable rule.

A single natural-language constraint may yield multiple AtomicConstraints.

Example:
“Total sleep must be at least 7 hours, and no wake period longer than 20 minutes.”
becomes two atomic constraints.

export interface AtomicConstraint {
  /**
   * Stable unique identifier.
   */
  id: string;

  /**
   * Original parent raw constraint text.
   */
  raw: string;

  /**
   * Optional normalized short form.
   * Example: "total_sleep_minutes >= 420"
   */
  normalized?: string;

  /**
   * Constraint family.
   */
  constraintClass: ConstraintClass;

  /**
   * Operator used to test the constraint.
   */
  operator: EvaluationOperator;

  /**
   * Key of the derived variable this constraint evaluates.
   * Example: total_sleep_minutes
   */
  variableKey: string;

  /**
   * Human-readable variable label.
   */
  variableLabel: string;

  /**
   * Primary threshold value if applicable.
   * Examples: 420, 20
   */
  threshold?: number | string | boolean;

  /**
   * Secondary threshold for range/window comparisons.
   */
  thresholdUpper?: number | string;

  /**
   * Units for thresholds.
   */
  units?: string;

  /**
   * Whether violation is immediately disqualifying.
   * Example: dropping an object may be hard-invalid.
   */
  hard: boolean;

  /**
   * Whether the constraint is required for admissibility
   * or advisory for margin shaping.
   */
  priority: "REQUIRED" | "PREFERRED";

  /**
   * Variable most likely to govern violation or margin.
   */
  decisiveVariableHint?: string;

  /**
   * Optional temporal context.
   * Example: pre-lift window, before frost, nightly sleep block
   */
  temporalContext?: {
    type: "INTERVAL" | "WINDOW" | "DEADLINE" | "SEQUENCE";
    reference?: string;
    start?: string | number;
    end?: string | number;
  };

  /**
   * Optional coupling to other constraints.
   */
  coupledConstraintIds?: string[];

  /**
   * Optional notes from decomposition.
   */
  notes?: string[];
}


⸻

MarginResult

Purpose

Represents distance from failure or constraint boundary.

This is the key type that turns categorical evaluation into continuous evaluation.

export interface MarginResult {
  /**
   * Key of the variable whose boundary is being measured.
   */
  variableKey: string;

  /**
   * Human-readable label.
   */
  variableLabel: string;

  /**
   * Operator being evaluated.
   */
  operator: EvaluationOperator;

  /**
   * Actual derived value.
   */
  actualValue: number | string | boolean;

  /**
   * Primary threshold value.
   */
  threshold?: number | string | boolean;

  /**
   * Secondary threshold for range-like cases.
   */
  thresholdUpper?: number | string;

  /**
   * Signed slack where possible.
   * Positive = safe/admissible side
   * Zero = boundary
   * Negative = violated side
   *
   * Example:
   * actual 450 min, threshold 420 min => +30
   * actual 390 min, threshold 420 min => -30
   */
  slack?: number;

  /**
   * Normalized margin score in [0.00, 1.00].
   * 1.00 = strongest margin
   * 0.00 = direct failure
   */
  normalizedScore: number;

  /**
   * Interpretable label.
   */
  band: "HIGH" | "MODERATE" | "LOW" | "FAILED";

  /**
   * Whether this margin is the limiting one among all constraints.
   */
  limiting: boolean;

  /**
   * Whether violation occurred.
   */
  violated: boolean;

  /**
   * Short formal explanation.
   */
  reason: string;

  /**
   * Optional suggested repair.
   */
  adjustments?: string[];
}
