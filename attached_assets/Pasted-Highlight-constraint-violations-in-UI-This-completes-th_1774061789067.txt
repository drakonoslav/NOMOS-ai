🧱 1. Extend render logic (StatusCard)

📁 StatusCard.tsx

Replace renderLine with multi-pass highlighting:

function renderLine(
  line: string,
  decisive?: string,
  status?: "LAWFUL" | "DEGRADED" | "INVALID"
) {
  let segments: React.ReactNode[] = [line];

  // 1. Decisive variable highlight (already exists)
  if (decisive) {
    segments = splitAndWrap(
      segments,
      decisive,
      "nm-decisive"
    );
  }

  // 2. Constraint violation (INVALID)
  if (status === "INVALID") {
    segments = splitAndWrapMulti(segments, [
      { key: "feasibility", cls: "nm-violation" },
      { key: "violation", cls: "nm-violation" },
      { key: "constraint", cls: "nm-violation" },
    ]);
  }

  // 3. Margin degradation (DEGRADED)
  if (status === "DEGRADED") {
    segments = splitAndWrapMulti(segments, [
      { key: "reduced", cls: "nm-degraded-highlight" },
      { key: "insufficient", cls: "nm-degraded-highlight" },
      { key: "degraded", cls: "nm-degraded-highlight" },
      { key: "margin", cls: "nm-degraded-highlight" },
    ]);
  }

  return segments;
}

🧠 2. Utility split functions

function splitAndWrap(
  nodes: React.ReactNode[],
  keyword: string,
  cls: string
): React.ReactNode[] {
  const regex = new RegExp(`(${keyword})`, "i");

  return nodes.flatMap((node, i) => {
    if (typeof node !== "string") return [node];

    const parts = node.split(regex);

    return parts.map((part, j) => {
      if (part.toLowerCase() === keyword.toLowerCase()) {
        return (
          <span key={`${i}-${j}`} className={cls}>
            {part}
          </span>
        );
      }
      return <span key={`${i}-${j}`}>{part}</span>;
    });
  });
}


⸻


function splitAndWrapMulti(
  nodes: React.ReactNode[],
  rules: { key: string; cls: string }[]
): React.ReactNode[] {
  let result = nodes;

  for (const rule of rules) {
    result = splitAndWrap(result, rule.key, rule.cls);
  }

  return result;
}
