🧠 3. Merge into single line

📁 validation_format.ts (new helper)

export function formatIssue(i: ValidationIssue): string {
  if (!i.example) return i.message;

  return `${stripPeriod(i.message)}; e.g., ${i.example}.`;
}

function stripPeriod(s: string): string {
  return s.replace(/\.$/, "");
}


⸻

🧩 4. Update ValidationPanel

📁 ValidationPanel.tsx

Replace rendering:

import { formatIssue } from "../validation/validation_format";

export function ValidationPanel({
  issues,
}: {
  issues: ValidationIssue[];
}) {
  if (!issues.length) return null;

  return (
    <div className="nm-validation">
      {issues.map((i) => (
        <div
          key={i.id}
          className={`nm-validation-item nm-${i.severity}`}
        >
          {formatIssue(i)}
        </div>
      ))}
    </div>
  );
}
