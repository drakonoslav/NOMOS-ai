🧠 2. Core detection logic

export function detectIssues(
  draft: Partial<NomosQuery>
): ValidationIssue[] {
  const issues: ValidationIssue[] = [];

  // ---- INTENT ----
  if (!draft.intent) {
    issues.push({
      id: "intent_missing",
      field: "intent",
      severity: "error",
      message: "Intent not declared.",
    });
  }

  // ---- CONSTRAINTS ----
  if (!draft.constraints || draft.constraints.length === 0) {
    issues.push({
      id: "constraints_missing",
      field: "constraint",
      severity: "error",
      message: "No constraints declared.",
    });
  } else {
    draft.constraints.forEach((c, i) => {
      if (!hasThreshold(c)) {
        issues.push({
          id: `constraint_threshold_${i}`,
          field: "constraint",
          severity: "error",
          message: "Constraint lacks threshold.",
        });
      }

      if (!hasUnit(c)) {
        issues.push({
          id: `constraint_unit_${i}`,
          field: "constraint",
          severity: "warning",
          message: "Constraint missing unit.",
        });
      }

      if (isAmbiguous(c)) {
        issues.push({
          id: `constraint_ambiguous_${i}`,
          field: "constraint",
          severity: "warning",
          message: "Constraint ambiguous.",
        });
      }
    });
  }

  return issues;
}


⸻

🧠 3. Detection helpers

function hasThreshold(text: string): boolean {
  return /[<>]=?|≥|≤|\d/.test(text);
}

function hasUnit(text: string): boolean {
  return /(h|hr|hours|min|kg|lbs|%)/i.test(text);
}

function isAmbiguous(text: string): boolean {
  const vague = ["better", "more", "less", "improve"];
  return vague.some((v) => text.toLowerCase().includes(v));
}

🧩 5. UI: ValidationPanel

📁 ValidationPanel.tsx

export function ValidationPanel({
  issues,
}: {
  issues: ValidationIssue[];
}) {
  if (!issues.length) return null;

  return (
    <div className="nm-validation">
      {issues.map((i) => (
        <div
          key={i.id}
          className={`nm-validation-item nm-${i.severity}`}
        >
          {i.message}
        </div>
      ))}
    </div>
  );
}

🧠 7. Block evaluation until valid

CRITICAL RULE

function canEvaluate(issues: ValidationIssue[]): boolean {
  return !issues.some((i) => i.severity === "error");
}
