function nowMs(): number {
  return Date.now();
}

function norm(v: number[]): number {
  return Math.sqrt(v.reduce((s, x) => s + x * x, 0));
}

function clamp(x: number, lo: number, hi: number): number {
  return Math.max(lo, Math.min(hi, x));
}

function meanControlValue(controlSequence: number[][]): number {
  if (controlSequence.length === 0) return 0;
  const total = controlSequence.reduce((sum, step) => sum + (step[0] ?? 0), 0);
  return total / controlSequence.length;
}

/**
 * One-step nominal roll-out helper for turning provisional LLM control sketches
 * into CandidatePlan nominal summaries.
 */
function simulateNominalPlan(
  modelRegistry: ModelRegistry,
  x0: number[],
  r0: number[],
  disturbances: number[],
  controlSequence: number[][],
  t0Sec: number,
  dtSec: number
): { nominalX: number[]; nominalU: number[]; nominalR: number[] } {
  let x = [...x0];
  let r = [...r0];
  let t = t0Sec;

  for (const step of controlSequence) {
    const u = step;
    x = modelRegistry.predictNextState(x, u, r, disturbances, t, dtSec);

    /**
     * Tiny mock resource depletion:
     * fuel decreases with |u[0]|.
     * This keeps resources coupled to control and prevents "free thrust".
     */
    const fuelBurn = 0.08 * Math.abs(u[0] ?? 0) + 0.01;
    r = [Math.max(0, (r[0] ?? 0) - fuelBurn)];

    t += dtSec;
  }

  return {
    nominalX: x,
    nominalU: [meanControlValue(controlSequence)],
    nominalR: r,
  };
}

/**
 * Convert an LLM proposal into a FeasibilityInput derived from the simulated nominal plan.
 */
function makeFeasibilityInput(
  nominalX: number[],
  nominalU: number[],
  nominalR: number[],
  tSec: number,
  theta: Record<string, number>,
  currentDependencyStamp: DependencyStamp,
  solutionDependencyStamp: DependencyStamp,
  equalityConstraints: ConstraintDefinition[],
  inequalityConstraints: ConstraintDefinition[],
  terminalConstraints: ConstraintDefinition[],
  conservationConstraints: ConstraintDefinition[]
): FeasibilityInput {
  return {
    x: nominalX,
    u: nominalU,
    r: nominalR,
    t: tSec,
    xT: nominalX,
    rT: nominalR,
    theta,
    equalityConstraints,
    inequalityConstraints,
    terminalConstraints,
    conservationConstraints,
    resources: [{ name: "fuel", index: 0, lowerBound: 0 }],
    currentDependencyStamp,
    solutionDependencyStamp,
  };
}

const primaryModel: RegisteredModel = {
  signature: {
    id: "mock-primary",
    version: "1.0.0",
    delayAware: true,
    stateDim: 2,
    measurementDim: 2,
    parameterNames: ["dragCoeff", "idleDrain"],
    description: "Tiny mock mission model with explicit delay awareness.",
  },
  implementation: {
    predictNextState: (
      x: number[],
      u: number[],
      _r: number[],
      w: number[],
      theta: Record<string, number>,
      _t: number,
      dt: number
    ): number[] => {
      const position = x[0] ?? 0;
      const energy = x[1] ?? 0;
      const thrust = u[0] ?? 0;
      const externalDrag = w[0] ?? 0;
      const dragCoeff = theta.dragCoeff ?? 0.1;
      const idleDrain = theta.idleDrain ?? 0.05;

      const drag = dragCoeff * externalDrag;
      const newPosition = position + dt * (thrust - drag);
      const newEnergy = energy + dt * (-0.2 * Math.abs(thrust) - idleDrain);

      return [newPosition, newEnergy];
    },

    predictMeasurement: (
      x: number[],
      _u: number[],
      _r: number[],
      _w: number[],
      _theta: Record<string, number>,
      _t: number
    ): number[] => {
      return [x[0] ?? 0, x[1] ?? 0];
    },

    measurementJacobian: () => {
      return [
        [1, 0],
        [0, 1],
      ];
    },

    fisherInformationMinEigenvalue: () => 1.0,
  },
  fallbackModelId: "mock-fallback",
};

    delayAware: true,
    stateDim: 2,
    measurementDim: 2,

      x: number[],

      _r: number[],

const controlProposals = proposalBundle.proposals.filter(

const candidatePlans: CandidatePlan[] = controlProposals.map((proposal) => {
  const controlSequence = proposal.planSketch!.controlSequence;

  const nominal = simulateNominalPlan(
    modelRegistry,

    resourcesAtObserve,
    disturbancesAtObserve,
    controlSequence,
    measurement.timestamp / 1000,
    1.0

    nominal.nominalX,
    nominal.nominalU,
    nominal.nominalR,
    measurement.timestamp / 1000,

    currentDependencyStamp,
    solutionDependencyStamp,
    equalityConstraints,
    inequalityConstraints,
    terminalConstraints,
    conservationConstraints
  );

  const maxControl = Math.max(...controlSequence.map((s) => Math.abs(s[0] ?? 0)), 0);
  const sensitivityScale = 1 + maxControl * 6;

  return proposer.toCandidatePlan({
    proposal,
    nominalX: nominal.nominalX,
    nominalU: nominal.nominalU,
    nominalR: nominal.nominalR,
    feasibilityInput,
    robustnessConfig: {
      epsilonMin: 0.03,
      delayPresent: true,
      analyzedOnDelayedModel: true,
      sensitivityMatrix: [
        [1.1 * sensitivityScale, 0.1],
        [0.05, 1.0 + 0.5 * sensitivityScale],

What changed

The key change is that this section:
	•	used to hand-author planA and planB
	•	now calls LLMProposer.propose(...)
	•	filters for CONTROL_PLAN proposals
	•	simulates each proposed control sketch into a nominal trajectory
	•	wraps each into a CandidatePlan
	•	sends those to decision_engine.ts

So the runtime chain is now exactly:

belief → observer → model → llm_proposer → decision_engine → verification_kernel → constitution_guard → audit_log
